  async addTag(tagId: string): Promise<void> {

    });

      include: { tag: true },

  async removeTag(tagId: string): Promise<void> {
    await db.prisma.recordTag.deleteMany({
      where: {
        recordId: this.id,
        tagId: tagId,
      },
    });
  }
